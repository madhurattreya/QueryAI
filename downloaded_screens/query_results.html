Customer Churn Analysis

Analysis of Q3 churn metrics filtered by high-value accounts across NA and EMEA regions.

code

Generated SQL Query

WITH RankedCustomers AS (
    SELECT 
        customer_id,
        company_name,
        region,
        arr_usd,
        churn_probability_score,
        ROW_NUMBER() OVER(PARTITION BY region ORDER BY churn_probability_score DESC) as risk_rank
    FROM enterprise_data.customer_metrics_q3
    WHERE arr_usd > 100000
      AND region IN ('NA', 'EMEA')
      AND status = 'Active'
)
SELECT 
    customer_id,
    company_name,
    region,
    ROUND(arr_usd / 1000, 2) as arr_k,
    ROUND(churn_probability_score * 100, 1) as risk_pct,
    last_login_days_ago
FROM RankedCustomers c
LEFT JOIN enterprise_data.usage_logs u ON c.customer_id = u.account_id
WHERE risk_rank <= 50
ORDER BY risk_pct DESC;
