Here is the SQL query to calculate the churn rate grouped by plan tier over the last 6 months. It assumes a user is considered churned if `is_active` is false.

SQL 
WITH monthly_status AS (
    SELECT 
        plan_tier,
        COUNT(*) as total_users,
        SUM(CASE WHEN is_active = false THEN 1 ELSE 0 END) as churned_users
    FROM users_main
    WHERE signup_date >= CURRENT_DATE - INTERVAL '6 months'
    GROUP BY plan_tier
)
SELECT 
    plan_tier,
    total_users,
    churned_users,
    ROUND((churned_users::numeric / total_users) * 100, 2) as churn_rate_pct
FROM monthly_status
ORDER BY churn_rate_pct DESC;
